### why dynamic extraction is the critical case

```ts
// THE USE CASE: cache comes from context, type must enforce WithCacheUri
const _runPlay = async (
  input: { query: string },
  context: { cache: WithCacheUri<SimpleCache<string>> },  // <-- context MUST require WithCacheUri
): Promise<string> => fetchData(input.query);

const runPlay = withSimpleCacheAsync(_runPlay, {
  cache: ({ fromInput }) => fromInput[1].cache,  // extraction function

// if context.cache is SimpleCache (no uri), meta: 'include' MUST NOT compile
```

the type system must force `context.cache` to be `WithCacheUri` when `meta: 'include'` is used with dynamic extraction.

## summary

add `meta?: 'include' | 'exclude'` option to `withSimpleCacheAsync` that changes the return type:
- `'exclude'` (default): returns direct value (backwards compat)
- `'include'`: returns `{ output, cached: { uri: string } }` — requires cache with `uri` method

key deliverables:
- `WithCacheUri<TCache>` type — intersects any `SimpleCache` with `uri: (key: string) => string`
- `meta: 'include'` option — only available when cache has `uri` method (type + runtime enforced)
- `cached.uri` — the full URI where the cached value is stored

## implementation detail

### WithCacheUri type

```ts
/**
 * intersect any SimpleCache with uri resolution capability
 *
 * use this type when you need to require a cache that can expose its storage location
 */
export type WithCacheUri<TCache extends SimpleCache<any>> = TCache & {
  /**
   * compute the URI for a given cache key
   *
   * @returns the full URI where this key is stored (e.g., 's3://bucket/prefix/key')
   */
  uri: (key: string) => string;
};
```

**usage:**
```ts
// require cache with uri support in function signature
const downloadAndCache = async (
  input: { url: string },
  context: { cache: WithCacheUri<SimpleCache<string>> },  // <-- requires uri method
) => {
  const cached = withSimpleCacheAsync(fetchUrl, {
    cache: context.cache,
    meta: 'include',  // allowed because cache has uri
  });
  return cached(input);
};

// dynamic extraction — context type propagates through
const _fetchWithCache = async (
  input: { url: string },
  context: { artifact: { cache: WithCacheUri<SimpleCache<string>> } },  // <-- uri required in context type
): Promise<string> => {
  return fetch(input.url).then((r) => r.text());
};

const fetchWithCache = withSimpleCacheAsync(_fetchWithCache, {
  cache: ({ fromInput }) => fromInput[1].artifact.cache,  // extracts WithCacheUri<SimpleCache>
  meta: 'include',  // allowed because extracted cache has uri
  serialize: { key: (input) => input.url },
});

// caller provides cache with uri via context
const result = await fetchWithCache(
  { url: 'https://example.com' },
  { artifact: { cache: s3Cache } },  // s3Cache must satisfy WithCacheUri<SimpleCache>
);
console.log(result.cached.uri);  // 's3://bucket/prefix/https%3A%2F%2Fexample.com'

// type error if cache lacks uri
const badExample = async (
  input: { url: string },
  context: { cache: SimpleCache<string> },  // <-- no uri method
) => {
  const cached = withSimpleCacheAsync(fetchUrl, {
    cache: context.cache,
    meta: 'include',  // ❌ type error: meta 'include' not allowed
  });
};
```

### WithSimpleCacheAsyncOptions extension

```ts
export interface WithSimpleCacheAsyncOptions<L extends AsyncLogic, C extends SimpleCache<any>> {
  // ... extant options ...

  /**
   * whether to include cache metadata in the return value
   *
   * - 'exclude' (default): returns the logic's return value directly
   * - 'include': returns { output, cached } — only available when cache has uri method
   */
  meta?: 'include' | 'exclude';
}
```

### conditional return type

```ts
/**
 * detect if cache has uri method
 */
type HasCacheUri<C> = C extends { uri: (key: string) => string } ? true : false;

/**
 * allowed meta values depend on whether cache has uri method
 */
type MetaValue<C extends SimpleCache<any>> =
  HasCacheUri<C> extends true
    ? 'include' | 'exclude'
    : 'exclude';  // 'include' not available without uri

/**
 * return type depends on meta value
 */
type WithMetaReturn<L extends AsyncLogic, M extends 'include' | 'exclude' | undefined> =
  M extends 'include'
    ? (...args: Parameters<L>) => Promise<{
        output: Awaited<ReturnType<L>>;
        cached: { uri: string };
      }>
    : L;

/**
 * options interface with conditional meta
 */
export interface WithSimpleCacheAsyncOptions<
  L extends AsyncLogic,
  C extends SimpleCache<any>,
  M extends MetaValue<C> | undefined = undefined,
> {
  cache: C | SimpleCacheExtractionMethod<Parameters<L>, C>;
  serialize?: { ... };
  deserialize?: { ... };
  expiration?: UniDuration;
  bypass?: { ... };
  meta?: M;
}

export const withSimpleCacheAsync = <
  L extends AsyncLogic,
  C extends SimpleCache<any>,
  M extends MetaValue<C> | undefined = undefined,
>(
  logic: L,
  options: WithSimpleCacheAsyncOptions<L, C, M>,
): WithMetaReturn<L, M> => { ... }
```

**type-level restriction:** `meta: 'include'` only compiles when `C` has `uri` method (detected via `HasCacheUri<C>`).

**runtime restriction:** guard for JS users or type assertions:
```ts
if (meta === 'include' && typeof cache.uri !== 'function') {
  throw new BadRequestError('meta: include requires cache with uri method', { cache });
}
```

### uri extraction

**prerequisite:** `simple-on-disk-cache` must be updated to return metadata from operations.

the cache is the source of truth for where items are stored — it knows the base path, may transform keys internally, and computes the full URI. the URI should come from the **operation result**, not from a property on the cache object.

**proposed `simple-on-disk-cache` change:**

```ts
// current interface — unchanged
set: (key, value, options) => Promise<void>;
get: (key) => Promise<string | undefined>;

// new optional method — caches that support URIs expose this
uri?: (key: string) => string;
```

caches compute the URI from their config + key. example implementation:

```ts
// in simple-on-disk-cache
uri: (key: string) => {
  if (isS3Directory(directory))
    return `s3://${directory.s3.bucket}/${directory.s3.prefix}/${key}`;
  if (isMountedDirectory(directory))
    return `file://${directory.mounted.path}/${key}`;
  return undefined;
}
```

**wrapper extraction (applies to both `withSimpleCache` and `withSimpleCacheAsync`):**

```ts
// only compute uri when meta: 'include'
if (meta === 'include') {
  const cachedUri = typeof cache.uri === 'function' ? cache.uri(key) : undefined;
  return { output: deserializedValue, cached: { uri: cachedUri } };
}
return deserializedValue;
```

**backwards compatibility:** caches without `uri` method work unchanged — `cachedUri` is simply undefined.

**note:** wrap logic must apply to ALL return paths in withSimpleCacheAsync:
- bypass.set early return (line ~191)
- cache hit return (line ~198)
- cache miss return (line ~203)

each path wraps when `meta === 'include'`.
